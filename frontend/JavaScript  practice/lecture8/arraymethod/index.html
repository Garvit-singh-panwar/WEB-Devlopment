forEach

arr.forEach(some function defination or name)

        let arr = {1,2,3,4,5};
        let print = function (el) {
            console.log(el);
        };

        arr.forEach(print);
        arr.foreach(el) => {
            console.log(el);
        };

Map

let newArr = arr.map(some function defination or name);

        // map function creates a new array with returned value 
        let num = [1,2,3,4]; 
        let double = num.map(function(el){
            return el*2;
        });

        // if callback is false it do not add element in new array

Every

returns true if every element of array gives true for some function . Else return false.

arr.every(some function defination or name);

        [1,2,3,4].eevery( (el) => (el%2 == 0));
        //false
        [2,4].every( (el) =>  (el%2 == 0));
        //true
        It is like a logical AND

Some

Returns true if some elements of array give true for some function Else returns false .

arr.some(some function defination or name);

        [1,2,3,4].some( (el) => (el%2 == 0));
        //true
        [1,3].some( (el) => (el%2 == 0));
        //false
        // it is like a logical OR

Reduce

Reduces the array to a single value

arr.reduce(reducer function with 2 variables for (accumulator , element) );

        // every element is passed in element variable
        // returned value is stored in accumulator
        [1,2,3,4].reduce( (res,el) => (res+el) );
        //10
