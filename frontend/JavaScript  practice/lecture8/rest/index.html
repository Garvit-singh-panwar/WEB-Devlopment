Rest

Allows a function to take an indefinite number of arguments and bundles them in an array


            function sum(...args) {
                return args.reduce( (add,el) => add +el);
            }

            function sum(...args)
            {
                return args.reduce((sum,el) => sum+el);
            }

            function min(...args){
                return args.reduce((min,el) => 
                {
                    if(min > el)
                    {
                        return el;
                    }
                    else
                    {
                        return min;
                    }
                  
                });
            }
